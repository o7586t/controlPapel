    Esta es la primera clase que diseño en ruby, y gestiona la cantidad de papel que tenemos almacenada para su uso.

def comprobacionCantidad(cantidad)
  cifra=true
  if ((cantidad < 0) or (cantidad > 999999)) then
    puts "La cifra introducida es incorrecta"
    cifra=false
  end
  return cifra
end
